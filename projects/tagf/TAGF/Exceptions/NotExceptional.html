def initialize(*args, **kwargs)
  _dbg_exception_start(__callee__)
  super
  if (@msg.nil?)
    offender	= kwargs[:offender] || args[0]
    @msg		= format('not an exception or ' +
                           'exception class: %s:%s',
                           offender.class.name,
                           offender.inspect)
  end
  self.set_message(@msg)
end
